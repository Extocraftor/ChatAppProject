   236	        # channel_id -> user_id -> websocket

   243	    async def connect(self, websocket: WebSocket, channel_id: int, user_id: int, username: str):
   244	        await websocket.accept()
   245	        if channel_id not in self.active_connections:
   246	            self.active_connections[channel_id] = {}
   247	        if channel_id not in self.usernames:
   248	            self.usernames[channel_id] = {}
   249	        if channel_id not in self.mute_states:
   250	            self.mute_states[channel_id] = {}
   251	
   252	        # If the same user reconnects quickly, close the stale socket first.
   253	        existing_socket = self.active_connections[channel_id].get(user_id)
   254	        if existing_socket is not None and existing_socket is not websocket:
   255	            try:
   256	                await existing_socket.close(code=1000)
   257	            except Exception:
   258	                pass
   259	
   260	        self.active_connections[channel_id][user_id] = websocket
   261	        self.usernames[channel_id][user_id] = username
   262	        self.mute_states[channel_id].setdefault(user_id, False)
   263	
   264	    def disconnect(self, channel_id: int, user_id: int, websocket: WebSocket | None = None) -> bool:
   265	        removed = False
   266	        if channel_id in self.active_connections:
   267	            channel_connections = self.active_connections[channel_id]
   268	            current_socket = channel_connections.get(user_id)
   269	            # Ignore stale-disconnect callbacks from sockets that were replaced.
   270	            if websocket is not None and current_socket is not websocket:
   271	                return False
   272	
   273	            if user_id in channel_connections:
   274	                channel_connections.pop(user_id, None)
   275	                removed = True
   276	            if not channel_connections:
   277	                self.active_connections.pop(channel_id, None)
   278	
   279	        if channel_id in self.usernames:
   280	            self.usernames[channel_id].pop(user_id, None)
   281	            if not self.usernames[channel_id]:
   282	                self.usernames.pop(channel_id, None)
   283	
   284	        if channel_id in self.mute_states:
   285	            self.mute_states[channel_id].pop(user_id, None)
   286	            if not self.mute_states[channel_id]:
   287	                self.mute_states.pop(channel_id, None)
   288	        return removed

====================================================================
SECTION D - HOW THIS WAS PLANNED TO WORK (INTENDED FLOW)
====================================================================
1) A user types a normal chat message like: play <url or song name>.
2) Backend text websocket (`/ws/{channel_id}/{user_id}`) always runs `_handle_music_play_command(...)` for new messages.
3) `_handle_music_play_command`:
   - Parses the command.
   - Verifies the user is currently in a voice channel (`voice_manager.find_channel_for_user`).
   - Resolves a playable stream URL via yt-dlp (with multiple strategies and optional cookie sources).
   - Wraps the stream URL with `/audio-proxy` when possible.
   - Broadcasts a `music_play` event to everyone in that voice channel.
   - Broadcasts a `music_bot_notice` in text chat (e.g., “Now playing: ...”).
4) Frontend receives:
   - `music_bot_notice` on the text websocket, shown as a local chat message from "Music Bot".
   - `music_play` on the voice websocket, then `_handleMusicPlaySignal` starts playback using `audioplayers` (`setSource(UrlSource(...))` + `resume()`).
5) When leaving voice channel or disposing app state, frontend stops/disposes the music player.

Implementation notes visible from code:
- Spotify links are converted to yt-dlp search (`ytsearch1:<spotify_url>`) rather than direct Spotify streaming.
- SoundCloud gets a custom format preference to avoid manifest/HLS when possible.
- `/audio-proxy` streams remote audio bytes to clients and forwards useful headers.
- `spotipy` exists in requirements but is not used in the shown backend music command path.
